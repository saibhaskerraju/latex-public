    colorlinks=true, % Enable colored links  
    linkcolor=myhyperlinkcolor, % Color of internal links (sections, pages, etc.)  
    filecolor=myhyperlinkcolor, % Color of file links  
    urlcolor=myhyperlinkcolor, % Color of URLs  
    citecolor=myhyperlinkcolor, % Color of citations  
    pdfborderstyle={/S/U/W 1} % Underline style for all links  
}
